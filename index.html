local DataStoreService = game:GetService("DataStoreService")
local Players = game:GetService("Players")

local DataStoreUtils = require(game.ReplicatedStorage.DataStoreUtils)
local ItemInfo = require(game.ReplicatedStorage.ItemInfoScript)

local DataStoreHighestScore = DataStoreService:GetDataStore("DataStoreHighestScore")

local DataManager = require(script.Parent.Data.DataManager)

local clickerFunction = Instance.new("RemoteFunction")
local purchaseFunction = Instance.new("RemoteFunction")
local resetFunction = Instance.new("RemoteFunction")

clickerFunction.Name = "ClickerFunction"
purchaseFunction.Name = "PurchaseFunction"
resetFunction.Name = "ResetFunction"

clickerFunction.Parent = game.ReplicatedStorage
purchaseFunction.Parent = game.ReplicatedStorage
resetFunction.Parent = game.ReplicatedStorage

local highestScoreRemote = Instance.new("RemoteEvent")
highestScoreRemote.Name = "HighestScoreRemote"
highestScoreRemote.Parent = game.ReplicatedStorage

function LoadHighestScore(retryCount: number): number
	local result, _ = DataStoreUtils.Action(retryCount, function()
		return DataStoreHighestScore.GetAsync(DataStoreHighestScore, "HighestScore")
	end)

	if result == nil then
		return 0
	end

	return result
end

function SaveHighestScore(value: number, retryCount: number): string
	local _, err = DataStoreUtils.Action(retryCount, function()
		return DataStoreHighestScore.SetAsync(DataStoreHighestScore, "HighestScore", value)
	end)

	return err
end

local currentHighestScore = LoadHighestScore(nil)

clickerFunction.OnServerInvoke = function(player)
	local value = 0
	
	for itemName, itemValue in pairs(ItemInfo.ItemValue) do
		local itemCount = DataManager.GetData(player, { "Items", itemName })
		if itemCount == nil then itemCount = 0 end
		
		value += itemCount * itemValue
	end
	
	value += 1

	DataManager.AddToData(player, {"Currency"}, value, true)
	
	local currency = DataManager.GetData(player, {"Currency"})
	
	if currency > currentHighestScore then
		currentHighestScore = currency
	end
end

purchaseFunction.OnServerInvoke = function(player, itemName)
	if itemName == nil then
		warn("error: invalid item", itemName)
		return
	end
	
	local itemCount = DataManager.GetData(player, { "Items", itemName })
	local price = ItemInfo.GetItemPrice(player, itemName, itemCount)
	
	if price == nil then
		warn("error: invalid item", itemName)
		return
	end
	
	local currency = DataManager.GetData(player, {"Currency"})
	
	if currency < price then
		return
	end
	
	DataManager.AddToData(player, {"Currency"}, -price, true)
	DataManager.AddToData(player, { "Items", itemName }, 1, true)
end

resetFunction.OnServerInvoke = function(player)
	DataManager.SetData(player, {"Currency"}, 0, true)
	
	for itemName, _ in pairs(ItemInfo.ItemValue) do
		DataManager.SetData(player, { "Items", itemName }, 0, true)
	end
end

local oldHighestScore = 0

while true do
	if currentHighestScore > oldHighestScore then
		oldHighestScore = currentHighestScore
		SaveHighestScore(currentHighestScore)
	end

	highestScoreRemote:FireAllClients(currentHighestScore)

	task.wait(1)
end

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local ServerScriptService = game:GetService("ServerScriptService")

local ProfileStore = require(game.ReplicatedStorage.Libraries.ProfileStore)
local Replica = require(game.ReplicatedStorage.Libraries.ReplicaServer)

local DataManager = require(ServerScriptService.Data.DataManager)

local PlayerTemplate = require(ServerScriptService.Data.PlayerTemplate)

function GetStoreName()
	if RunService:IsStudio()  then
		return "Test"
	else
		return "Live"
	end
end

local PlayerStore = ProfileStore.New(GetStoreName(), PlayerTemplate)

local Token = Replica.Token("PlayerData")

function PlayerAdded(player: Player)
	-- Profile store
	local playerProfile = PlayerStore:StartSessionAsync("Player_" .. player.UserId, {
		Cancel = function()
			return player.Parent ~= Players
		end
	})
	
	if playerProfile == nil then
		player:Kick("Couldn't load data. Please rejoin.")
		return
	end
	
	playerProfile:AddUserId(player.UserId)
	playerProfile:Reconcile()
	
	playerProfile.OnSessionEnd:Connect(function()
		DataManager.Profiles[player] = nil
		player:Kick("Couldn't load data. Please rejoin.")
	end)
	
	if player.Parent ~= Players then
		playerProfile:EndSession()
		return
	end
	
	DataManager.Profiles[player] = playerProfile

	-- Replica
	local replica = Replica.New({
		Token = Token,
		Data = playerProfile.Data,
	})

	DataManager.Replicas[player] = replica

	replica:Replicate()
end

for _, player in Players:GetPlayers() do
	task.spawn(PlayerAdded, player)
end

Players.PlayerAdded:Connect(PlayerAdded)

Players.PlayerRemoving:Connect(function(player)
	local profile = DataManager.Profiles[player]
	if profile == nil then return end
	
	profile:EndSession()
	DataManager.Profiles[player] = nil
end)
